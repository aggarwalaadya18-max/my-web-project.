// Lab 4 - Prompt Quizzer // Tech-themed questions, still beginner-level :) let quizQuestions = [ { question: "What does CPU stand for?", answer: "central processing unit" }, { question: "Which company created the Windows operating system?", answer: "microsoft" }, { question: "What does 'www' stand for in a website URL?", answer: "world wide web" }, { question: "Which programming language is primarily used for styling web pages?", answer: "css" }, { question: "What is the name of the computer network that became the modern internet?", answer: "arpanet" } ]; function runQuiz() { let score = 0; for (let i = 0; i < quizQuestions.length; i++) { let userAnswer = prompt(quizQuestions[i].question); if (userAnswer) { userAnswer = userAnswer.toLowerCase().trim(); } else { userAnswer = ""; } if (userAnswer === quizQuestions[i].answer) { score++; alert("Correct! :)"); } else { alert("Wrong! The correct answer was: " + quizQuestions[i].answer); } } alert("Quiz finished!\nYou scored " + score + " out of " + quizQuestions.length); } runQuiz();
